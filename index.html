const qsort = ([pivot, ...rest]) =>
    rest.length
    ? [
        ...qsort(filter(rest, x => x <= pivot)),
        pivot,
        ...qsort(filter(rest, x => x > pivot))
    ]
    : typeof pivot !== 'undefined' ? [pivot] : [];

const reduce = ([head, ...rest], f, base) =>
    typeof head !== 'undefined'
    ? reduce(rest, f, f(base, head))
    : base;

const map = (arr, f) => reduce(arr, (acc, x) => [...acc, f(x)], []);
